const express = require('express'); const http = require('http'); const socketIo = require('socket.io'); const app = express(); const server = http.createServer(app); const io = socketIo(server); app.use(express.static('.')); // Sirve archivos estáticos (HTML, CSS, JS) io.on('connection', (socket) => { console.log('Usuario conectado:', socket.id); // Manejar mensajes de chat socket.on('chat message', (msg) => { io.emit('chat message', msg); // Broadcast a todos }); // Manejar typing socket.on('typing', (data) => { socket.broadcast.emit('typing', data); // Envía a todos excepto el emisor }); socket.on('stop typing', (data) => { socket.broadcast.emit('stop typing', data); }); socket.on('disconnect', () => { console.log('Usuario desconectado:', socket.id); }); }); const PORT = 3000; server.listen(PORT, () => { console.log(`Servidor corriendo en http://localhost:${PORT}`); });
